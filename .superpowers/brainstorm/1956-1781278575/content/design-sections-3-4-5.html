Section 3 — Queue & Worker
// worker/index.ts — starts with: npm run worker
const boss = new PgBoss(DATABASE_URL)
await boss.start()
await boss.work('image-job', { teamSize: 5 }, imageWorker)
await boss.work('video-job', { teamSize: 2 }, videoWorker)

// image-worker.ts
if (job.data.provider === 'runware') → providers/runware.ts
if (job.data.provider === 'vertex')  → providers/vertex-image.ts
→ download image → uploads/images/:projectId/:index.png
→ UPDATE jobs SET status='completed', output_path=…
→ UPDATE projects SET completed_jobs += 1

// video-worker.ts — long-running, polls Vertex AI
→ POST to Vertex Veo → get operation_id → save to DB
→ poll every 30s → when done → download MP4
→ UPDATE jobs SET status='completed', output_path=…

// Retry: pg-boss retries failed jobs up to 3x automatically
// Concurrency: 5 image workers, 2 video workers in parallel
Section 4 — API Routes
POST /api/projects               create project
GET    /api/projects               list all projects
GET    /api/projects/:id           project + jobs
DELETE /api/projects/:id           delete project + files

POST /api/projects/:id/generate  submit prompts → enqueue jobs
POST /api/projects/:id/pause     pause queue
POST /api/projects/:id/resume    resume queue
POST /api/projects/:id/cancel    cancel all pending jobs
POST /api/projects/:id/retry     retry all failed jobs

GET    /api/projects/:id/status    polled every 2s for live progress
GET    /api/projects/:id/download  stream ZIP of all outputs

PATCH /api/jobs/:id               edit prompt on failed job
POST /api/jobs/:id/retry         retry single job

Error Handling
  • pg-boss auto-retries failed jobs up to 3 times with exponential backoff
  • Failed jobs show error message + Edit Prompt + Retry buttons
  • Vertex video: if operation_id saved and server restarts, worker resumes polling on startup
  • Runware API errors: stored in error_message, surfaced in UI per job
Testing
  • Unit tests for provider modules (mock API responses)
  • Integration tests for API routes using a test PostgreSQL DB
  • Manual E2E: generate 5-image project, verify files appear on disk, download ZIP
